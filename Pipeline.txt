pipeline {
    agent any

    environment {
        CONFIGURATION = "Release"
        ALLURE_RESULTS = "allure-results"
        ALLURE_REPORT = "allure-report"
        PLAYWRIGHT_BROWSERS_PATH = "C:\\playwright-cache"
    }

    stages {

        stage('Checkout') {
            steps {
                git 'https://github.com/sauza2025-max/PlaywrightTests.git'
            }
        }

        stage('Clean Workspace') {
            steps {
                bat '''
                    if exist allure-results rmdir /s /q allure-results
                    if exist allure-report rmdir /s /q allure-report
                '''
            }
        }

        stage('Restore') {
            steps {
                bat 'dotnet restore'
            }
        }

        stage('Build') {
            steps {
                bat 'dotnet build --configuration %CONFIGURATION% --no-restore'
            }
        }

        stage('Install Playwright Browsers (cached)') {
            steps {
                bat '''
                    if not exist C:\\playwright-cache\\chromium (
                        dotnet tool install --global Microsoft.Playwright.CLI
                        playwright install
                    ) else (
                        echo Playwright cache found - skipping install
                    )
                '''
            }
        }

        stage('Test Execution') {
            steps {
                bat '''
                    dotnet test --no-build ^
                    --configuration %CONFIGURATION% ^
                    --results-directory %ALLURE_RESULTS% ^
                    --filter "Category=Login"

                    set TEST_EXIT=%ERRORLEVEL%
                    echo Test Exit Code: %TEST_EXIT%
                '''
            }
        }

        stage('Generate Allure Report') {
            steps {
                bat '''
                    allure generate allure-results --clean -o allure-report
                    exit /b 0
                '''
            }
        }
    }

    post {

        always {
            script {
                try {
                    allure results: [[path: 'allure-results']]
                } catch (err) {
                    echo "Allure publish failed but build continues: ${err}"
                }

                archiveArtifacts artifacts: 'allure-report/**', fingerprint: true
            }
        }

        unsuccessful {
            unstable('⚠️ Some tests failed (check Allure report)')
        }

        cleanup {
            bat '''
                if exist allure-results rmdir /s /q allure-results
                if exist allure-report rmdir /s /q allure-report
            '''
        }
    }
}
